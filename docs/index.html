Simple API

Stream format: [4-byte size LE][FlatBuffer with file_id]...

// Create database from schema
const db = new FlatSQL.FlatSQLDatabase(schema);

// Register file identifiers for routing
db.registerFileId("USER", "User");
db.registerFileId("POST", "Post");

// Stream size-prefixed FlatBuffers
const bytesConsumed = db.ingest(streamData);

// Or ingest a single FlatBuffer
const rowid = db.ingestOne(flatbuffer);

// Query with SQL
const result = db.query("SELECT * FROM User WHERE age > 25");
